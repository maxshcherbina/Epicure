104        """ Visibility of the parameters in the GUI """

106            get_files.segmentation_file.visible = False
107            get_files.segment_with_epyseg.visible = False
108        advanced_visibility()

111        """ Handle the visibility of the advanced parameters """
112        get_files.output_dirname.visible = get_files.advanced_parameters.value
113        get_files.show_other_chanels.visible = get_files.advanced_parameters.value
114        get_files.process_frames_parallel.visible = get_files.advanced_parameters.value
115        get_files.nbparallel_threads.visible = get_files.advanced_parameters.value
116        get_files.junction_half_thickness.visible = get_files.advanced_parameters.value
117        get_files.verbose_level.visible = get_files.advanced_parameters.value
118        get_files.allow_gaps.visible = get_files.advanced_parameters.value
119        get_files.show_scale_bar.visible = get_files.advanced_parameters.value
120        get_files.epithelial_cells.visible = get_files.advanced_parameters.value

122    def load_movie_from_layers(movie, movie_path, segmented):
123        """ Load and display the selected layers """
124        start_time = ut.start_time()
125        nonlocal caxis, cval
126        caxis, cval = Epic.movie_from_layer(movie, movie_path)
127        imgdir = ut.get_directory(movie_path)
128        #print(imgdir)
129        
130        if segmented is None:
131            get_files.segmentation_file.visible = True
132            get_files.segmentation_file.value = pathlib.Path(imgdir)
133            get_files.segment_with_epyseg.visible = True
134        else:
135            get_files.segmentation_file.visible = False
136            get_files.trackmate_file.visible = False
137            get_files.geff_file.visible = False
138            get_files.segment_with_epyseg.visible = False
139        
140        labname = Epic.suggest_segfile( get_files.output_dirname.value )
141        Epic.set_names( get_files.output_dirname.value )
142        if labname is not None:
143            get_files.segmentation_file.value = pathlib.Path(labname)
144            Epic.read_epicure_metadata()    
145        if caxis is not None:
146            get_files.junction_chanel.max = cval-1
147            get_files.junction_chanel.visible = True
148            set_chanel()
149        show_metatdata(show=True)
150        get_files.allow_gaps.value = bool(Epic.epi_metadata["Allow gaps"])
151        get_files.verbose_level.value = int(Epic.epi_metadata["Verbose"])
152        get_files.call_button.enabled = True
153        if "MainChannel" in Epic.epi_metadata:
154            get_files.junction_chanel.value = int(Epic.epi_metadata["MainChannel"])
155
156    def show_metatdata(show=True):
157        """ Show or update the metadata parameters """
158        get_files.scale_xy.value = Epic.epi_metadata["ScaleXY"]
159        get_files.timeframe.value = Epic.epi_metadata["ScaleT"]
160        get_files.unit_xy.value = Epic.epi_metadata["UnitXY"]
161        get_files.unit_t.value = Epic.epi_metadata["UnitT"]
162        get_files.scale_xy.visible = show 
163        get_files.unit_xy.visible = show
164        get_files.timeframe.visible = show 
165        get_files.unit_t.visible = show

186        get_files.segment_with_epyseg.visible = True
187        get_files.allow_gaps.value = bool(Epic.epi_metadata["Allow gaps"])

190            get_files.junction_chanel.value = int(Epic.epi_metadata["MainChannel"])
191        get_files.call_button.enabled = True
192        ut.show_duration(start_time, header="Movie loaded in ")
193
194    def show_others():
195        """ Display other chanels from the initial movie """
196        for ochan in range(cval):
197            ut.remove_layer(viewer, "MovieChannel_"+str(ochan))
198        if get_files.show_other_chanels.value == True:
199            Epic.add_other_chanels(int(get_files.junction_chanel.value), caxis)

202        """ Set the correct chanel that contains the junction signal """

212    def launch_napari_epyseg():
213        """ Open napari-epyseg plugin to segment the intensity channel movie """
214        print("Running EpySeg with default parameters on the movie. To change the settings, use the napari-epyseg plugin outside of EpiCure or EpySeg module directly")
215        parameters = {"tile_width":256, "tile_height":256, "overlap_width":32, "overlap_height":32, "model":"epyseg default(v2)", "norm_min":0, "norm_max":1}
216        ut.show_progress( viewer, True )
217        progress_bar = progress( len(Epic.img) )
218        progress_bar.set_description( "Running epyseg on all frames..." )
219        progress_bar.update(0)
220        try:
221            from epicure.appose_epyseg import go_epyseg
222            class LogHandler(logging.Handler):
223                def emit(self, record):
224                    msg = self.format(record)
225                    progress_bar.set_description( msg )
226
227            def setup_logger( name="epyseg_seg" ):
228                logger = logging.getLogger(name)
229                handler = LogHandler()
230                formatter = logging.Formatter('[EpiCure] %(message)s')
231                handler.setFormatter( formatter )
232                logger.addHandler(handler)
233                logger.setLevel( logging.INFO )
234                return logger
235
236            logger = setup_logger()
237            segres = go_epyseg( Epic.img, parameters, progress_bar=None, logger=logger )
238        except Exception as e:
239            ut.show_error( "This option requires the plugin napari-epyseg that is missing.\nInstall it and restart" )
240            print(e)
241            return
242        ut.show_progress( viewer, False )
243        segname = str(get_files.image_file.value)+"_epyseg.tif"
244        ut.writeTif( segres, segname, 1.0, "uint8", what="Epyseg results saved in " )
245        get_files.segmentation_file.value = segname
246        get_files.segment_with_epyseg.visible = False

269    def get_files( 
270                   junction_chanel = 0,
271                   _ = "Image metadata",
272                   scale_xy = 1,
273                   unit_xy = "um",
274                   timeframe = 1,
275                   unit_t = "min",
276                   __ = "\nSegmentation",
277                   ___ = "Load from skeleton or label file",
278                   segmentation_file = pathlib.Path(cdir),
279                   ____ =  "OR\tfrom TrackMate\t",
280                   trackmate_file = pathlib.Path(cdir),
281                   _____ = "OR\tfrom GEFF   \t",
282                   geff_file = pathlib.Path(cdir),
283                   ______ = "OR \t\t\t",
284                   segment_with_epyseg = False,
285                   ________ = "\n",
286                   advanced_parameters = False,
287                   show_other_chanels = True,
288                   show_scale_bar = True,
289                   allow_gaps = True,
290                   epithelial_cells = True,
291                   process_frames_parallel = False,
292                   nbparallel_threads = ncpus,
293                   junction_half_thickness = 1,
294                   output_dirname = "epics",
295                   verbose_level = 1,
296                   go_help = False,
297                   ):
298
299        print("Starting")
300        imname, imdir, outdir = ut.extract_names( os.path.abspath(raw_movie_path), output_dirname )
301        update_save_history(imdir)
302        #ut.remove_widget(viewer, "Start EpiCure (epicure)")
303        ut.remove_all_widgets( viewer )
304        Epic.process_parallel = process_frames_parallel
305        Epic.set_verbose( verbose_level )
306        Epic.nparallel = nbparallel_threads
307        #Epic.check_shape()
308        Epic.set_thickness( junction_half_thickness )
309        Epic.set_scales(scale_xy, timeframe, unit_xy, unit_t)
310        Epic.set_scalebar( show_scale_bar )
311        Epic.set_gaps_option( allow_gaps )
312        Epic.set_epithelia( epithelial_cells )
313        ## to handle segmentation from layer or from file
314        seg_file = segmentation_file
315        ## check if another segmentation import has been loaded (otherwise they are default to a directory)
316        if trackmate_file.exists() and trackmate_file.is_file() and trackmate_file.suffix == ".xml": 
317            seg_file = trackmate_file
318        elif geff_file.suffix == ".geff" and geff_file.exists():
319            seg_file = geff_file
320        segmentation_input = {"File":seg_file} 
321        if segmented is not None:
322            segmentation_input["Layer"]=segmented
323
324        Epic.go_epicure(outdir, segmentation_input)
325    
326    set_visibility()
327    get_files.call_button.enabled = False
328    get_files.junction_chanel.visible = False
329    ## load movie if there is a path but nothing loaded yet
330    if (raw_movie is None) and (raw_movie_path is not None):
331        load_movie( raw_movie_path )
332    elif raw_movie is not None:
333        load_movie_from_layers( raw_movie, raw_movie_path, segmented )
334        show_metatdata(show=True)
335        if segmented is not None:
336            get_files.call_button.enabled = True
337    get_files.advanced_parameters.clicked.connect(set_visibility)
338    get_files.show_other_chanels.clicked.connect(show_others)
339    get_files.junction_chanel.changed.connect(set_chanel)
340    get_files.segment_with_epyseg.clicked.connect( launch_napari_epyseg )
341    get_files.go_help.clicked.connect( show_documentation )
342    return get_files, Epic
